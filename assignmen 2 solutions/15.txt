#include<stdio.h>

void main()
{
    int i,j,n;
    printf("Enter the number of rows of pattern to print = ");
    scanf("%d",&n);


    for(i=0;i<=n;i++)
    {
        for(j=1;j<=((n*2)+1);j++)
        {
            if (j==(n+1) || (j>=(n+1)-i && j<=(n+1)+i))
                printf("*");
            else    
                printf(" ");    
        }
        printf("\n");
    }
}
