#include<stdio.h>

void main()
{
    long int a,bin=0,n,i=1,dec=0;
 
    printf("Enter a decimal number = ");
    scanf("%ld",&a);
    n=a;

    while (n>0)
    {
        bin = bin + (n%2)*i ;
        n=n/2; 
        i=i*10;
    }
    
    printf("\nBinary of decimal number %ld is %ld.",a,bin);

    i=1;
    printf("\nEnter a Binary number = ");
    scanf("%ld",&a);
    n=a;

    while(n>0)
    {
        dec = dec + n%10 *i;
        i=i*2;
        n=n/10;
    }

    printf("\nDecimal of Binary number %d is %d",a,dec);
}
