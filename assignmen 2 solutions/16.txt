#include<stdio.h>

void main()
{
    int i,j,k=0;

    for(i=0;i<=4;i++)
    {
        for(j=1;j<=9;j++)
        {       
            if(j>=(5-i) && j<=5)
            {
                k=k+1;
                printf("%d",k);
            }
            else if(j<=(5+i) && j>5)
            {
                k=k-1;
                printf("%d",k);
            }
            else 
                printf(" ");
            
        }
        printf("\n");
    }

}
